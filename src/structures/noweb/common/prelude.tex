%% colorized parts
\setkomafont{part}{\mypart}
\newcommand{\mypart}[1]{%
    \setcounter{section}{0}
    \huge\sffamily\bfseries%
    \color{white}%
    \setlength{\fboxsep}{0cm}%already boxed
    \colorbox{mygray!90}{%
        \begin{minipage}{\linewidth}%
            \vspace*{4pt}%Space before
            \hspace*{3pt}
            #1
            \vspace*{4pt}%Space after
        \end{minipage}%
        % \ihead{#1}
    }}
