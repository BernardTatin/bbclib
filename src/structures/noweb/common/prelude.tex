    \begin{minipage}{170mm}

%% colorized sections
\setkomafont{section}{\mysection}
\newcommand{\mysection}[1]{%
    \LARGE\sffamily\bfseries%
    \color{white}%
    \setlength{\fboxsep}{0cm}%already boxed
    \colorbox{orange!80}{%
        \begin{minipage}{\linewidth}%
            \vspace*{4pt}%Space before
            \hspace*{3pt}
            #1
            \vspace*{4pt}%Space after
        \end{minipage}%
        % \ohead{#1}
    }}
